    {% Tiene incisos

      \protected@edef\tmp@a{\theejercicio}% expand as far as we can
      \@xa\g@addto@macro\@xa\primerafila\@xa{\tmp@a}%
      \egroup% restore old value of \tmp@a, effect of \g@addto.. is

      \setcounter{contgrilla}{1}

        \stepcounter{contgrilla}
